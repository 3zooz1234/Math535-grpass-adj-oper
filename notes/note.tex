\section{Riesz Representation Theorem}

\begin{thm}
Let $H$ be a Hilbert space.
For any bounded linear functional $\varphi \in H^{*}$,
there exists a unique element $y \in H$ such that\\

$\varphi(x)=\langle y, x\rangle \quad$ for all $x \in H$ and $\|\varphi\|=\|y\|$\\

\end{thm}

\begin{cor}
Every bounded sequence in a Hilbert space $H$ containing a weakly convergent subsequence.
\end{cor}

\begin{cor}
Let $T: H \rightarrow H^{*}$ be a bounded linear operator on a Hilbert space $H$.
Assume that $T$ is strongly monotone, i.e.

$$
\langle T(x), x\rangle \geqslant \alpha\|x\|^{2} \quad \forall x \in H \quad(\alpha>0)
$$

then $T(x)=f, x \in H$, has a unique solution for all $f \in H^{*}$.
\end{cor}


\section{The Adjoint}

\begin{defn}[Adjoint Operator]
  Let $H$ and $G$ be Hilbert spaces and let $T: H \longrightarrow G$ be a bounded linear operator. An operator $T^{*}: G \rightarrow H$ such that
\begin{equation}
\langle T x, y\rangle_{G}=\left\langle x, T^{*}y\right\rangle_{H} \quad
\end{equation}
 for all  x $\in H, y \in G$, is called the adjoint of $T$.\label{eq:adjoint}
\end{defn}

\begin{prop}
For any bounded linear operator $T$ on a Hilbert space $H$, the adjoint operator $T^{*}$ exists and is unique.
\end{prop}

The defining relation of the adjoint \eqref{eq:adjoint} shows that the action of $A$ on $H_0$ is mimicked by the action of the adjoint $A^{*}$ on $H_1$; this mimicry is visible only through the applicable inner products, so the adjoint itself depends on these inner products. Loosely, if $A$ has some effect, $A^{*}$ preserves the geometry of that effect while acting with a reversed domain and codomain.


\begin{exm}[Adjoint Operators]
For any Hilbert space $H$, consider $T:H \rightarrow H$ given by $Tx = \alpha x$ for some scalar $\alpha$. For any $x$ and $y$ in $H$,
$$
\langle Tx, y \rangle = \langle \alpha x, y \rangle = \alpha \langle x, y \rangle = \langle x, \bar{\alpha} y \rangle
$$
where the first equality follows from the definition of $T$, the second from the linearity in the first argument of the inner product, and the third from the conjugate linearity in the second argument of the inner product. Upon comparison with \eqref{eq:adjoint}, the adjoint of $T$ is $T^{*}y = \bar{\alpha}y$. Simply, the adjoint of multiplication by a scalar is multiplication by the conjugate of the scalar.
\end{exm}


\begin{exm}
The matrix of the adjoint of a linear map on $\mathbb{R}^{n}$ with matrix $A$ is $A^{T}$ since 
$$x \cdot (Ay) = (A^{T}x) \cdot y.$$ 
In component notation, we have
$$
\sum_{i=1}^{n} x_{i} \left( \sum_{j=1}^{n} a_{ij}y_{j} \right) = \sum_{j=1}^{n} \left( \sum_{i=1}^{n} a_{ij}x_{i} \right) y_{j}.
$$
The matrix of the adjoint of a linear map on $\mathbb{C}^{n}$ with complex matrix $A$ is the Hermitian conjugate matrix, $A^{*} = \overline{A^{T}}$.
\end{exm}

\begin{exm}
Suppose that $S$ and $T$ are the right and left shift operators on the sequence space $l^{2}(\mathbb{N})$, defined by
$$
S(x_{1}, x_{2}, x_{3}, \dots) = (0, x_{1}, x_{2}, x_{3}, \dots), \quad T(x_{1}, x_{2}, x_{3}, \dots) = (x_{2}, x_{3}, x_{4}, \dots).
$$
Then $T=S^{*}$, since
$$
\langle x, Sy \rangle = x_{2}\overline{y_{1}} + x_{3}\overline{y_{2}} + x_{4}\overline{y_{3}} + \dots = \langle Tx, y \rangle.
$$
\end{exm}

\begin{exm}[Local Averaging and its Adjoint]
The operator $A$ given by
$$
(Ax)_{n} = \int_{n-1/2}^{n+1/2} x(t) dt, \quad n \in \mathbb{Z},
$$
takes local averages of the function $x(t)$ to yield a sequence $(Ax)_{n}$. $A$ is a bounded linear operator from $\mathcal{L}^{2}(\mathbb{R})$ to $l^{2}(\mathbb{Z})$. For the adjoint:
$$
\langle Ax, y \rangle_{l^{2}} = \sum_{n \in \mathbb{Z}} (Ax)_{n} y_{n}^{*} = \sum_{n \in \mathbb{Z}} \left( \int_{n-1/2}^{n+1/2} x(t) dt \right) y_{n}^{*} = \int_{-\infty}^{\infty} x(t) \left( \sum_{n \in \mathbb{Z}} y_{n} 1_{[n-1/2, n+1/2)}(t) \right)^{*} dt 
$$

$$
= \langle x, A^{*}y \rangle_{\mathcal{L}^{2}}.
$$
The adjoint of local averaging forms a piecewise-constant function:
$$
(A^{*}y)(t) = \sum_{n \in \mathbb{Z}} y_{n} 1_{[n-1/2, n+1/2)}(t).
$$
\end{exm}

\begin{thm}
Suppose that $A:\mathcal{H}\rightarrow\mathcal{H}$ is a bounded linear operator on a Hilbert space $\mathcal{H}$ with closed range. Then the equation $Ax=y$ has a solution for $x$ if and only if $y$ is orthogonal to $\ker A^{*}$.
\end{thm}

\begin{exm}
Consider the multiplication operator $M:L^{2}([0,1])\rightarrow L^{2}([0,1])$ defined by
$$
Mf(x)=xf(x).
$$
Then $M^{*}=M$, and $M$ is one-to-one, so every $g\in L^{2}([0,1])$ is orthogonal to $\ker M^{*}$; but the range of $M$ is a proper dense subspace of $L^{2}([0,1])$, so $Mf=g$ is not solvable for every $g\in L^{2}([0,1])$.
\end{exm}

\section{The Adjoint}

\begin{defn}
  Let $K$ and $G$ be Hilbert spaces and let $T: H \longrightarrow G$ be a linear operator. An operator $T^{*}: G \rightarrow H$ such that

$$
\langle T x, y\rangle_{G}=\left\langle x, T^{*}y\right\rangle_{H} \quad \text { for all } x \in \mathcal{H}, y \in G
$$

is called \textbf{the adjoint }of $T$.
\end{defn}

\begin{thm}
  For any linear operator $T: H \rightarrow G$. the adjoint $T^{*}$ exists and is unique.
\end{thm}
\begin{proof}
  \textbf{Existence of the Adjoint:} We would like to show that for all $y\in G$, there is $w\in H$ such that 
  $$\langle Tx,y\rangle = \langle x,w\rangle$$
  for all $x\in H$. Let
  $$
  \begin{aligned}
    f_{y}: H&\longrightarrow\mathbb{F} \\
           x&\longmapsto\langle Tx,y\rangle
  \end{aligned}
  $$
  This map is linear, since it is the composition of two linear maps, namely $T$ and $\langle\cdot,y\rangle$. Also, this map is bounded since
  $$
  \begin{aligned}
    |f_{y}(x)| = |\langle Tx,y\rangle| &\leq \lVert y\rVert\lVert Tx\rVert \\
                                       &\leq \lVert y\rVert\lVert T\rVert \lVert x\rVert
  \end{aligned}
  $$
  So, $f_{y}\in H^{*}$. By Reisz Representation Theorem, for some $w\in H$, we can write $f_{y}$ as
  $$
  \langle Tx,y\rangle = f_{y}(x) = \langle x,w\rangle
  $$
  \newline
  \textbf{Uniqueness of the Adjoint:} Let $S:H\to G$ and $S':H\to G$ be adjoints of $T$, i.e. for all $x,y\in G$
  $$
    \begin{aligned}
      \langle Tx,y\rangle &= \langle x,Sy\rangle \\
      \langle Tx,y\rangle &= \langle x,S'y\rangle
    \end{aligned}
  $$
  We then have
  $$
  \begin{aligned}
    \langle x,(S-S')y\rangle = \langle x,Sy\rangle-\langle x,S'y\rangle = \langle Tx,y\rangle-\langle Tx,y\rangle = 0 \\
  \end{aligned}
  $$
  This implies that $(S-S')y=0$, and hence, $S-S'=0\implies S=S'$.
\end{proof}

\begin{thm}
  Let $H,G$ be Hilbert spaces and let $T:H\to G$ and $S:H\to G$ be linear operator, then
  \begin{enumerate}
    \item $\left(T^{*}\right)^{*}=T$
    \item $(T+S)^{*} = T^{*} + S^{*}$
    \item $(\alpha T)^{*} = \bar{\alpha}T^{*}$
    \item $(TS)^{*} = S^{*}T^{*}$
    \item $T^{*}(\alpha u+\beta v) = \bar{\alpha}T^{*}(u)+\bar{\beta}T^{*}(v)$
  \end{enumerate}
\end{thm}
\begin{proof}
  \begin{enumerate}
    \item We have
      $$
      \begin{aligned}
        \langle\left(T^{*}\right)^{*}x,y\rangle = \overline{\langle y,\left(T^{*}\right)^{*}x\rangle} = \overline{\langle T^{*}y,x\rangle} = \langle x,T^{*}y\rangle = \langle Tx,y\rangle
      \end{aligned}
      $$
      Therefore, $\left(T^{*}\right)^{*}=T$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(T+S)^{*}y\rangle = \langle (T+S)x,y\rangle &= \langle Tx,y\rangle+\langle Sx,y\rangle\\
                                                              &= \langle x,T^{*}y\rangle+\langle x,S^{*}y\rangle \\
                                                              &= \langle x,(T^{*}+S^{*})y\rangle
      \end{aligned}
      $$
      Therefore, $(T+S)^{*} = T^{*}+S^{*}$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(\alpha T)^{*}y\rangle = \langle\alpha Tx,y\rangle = \alpha\langle Tx,y\rangle = \alpha\langle x,T^{*}y\rangle &= \alpha\overline{\langle T^{*}y,x\rangle} \\
                                                                                                                                 &= \overline{\bar{\alpha}\langle T^{*}y,x\rangle} \\
                                                                                                                                 &= \overline{\langle \bar{\alpha}T^{*}y,x\rangle} \\
                                                                                                                                 &= \langle x,\bar{\alpha}T^{*}y\rangle
      \end{aligned}
      $$
      Therefore, $(\alpha T)^{*} = \bar{\alpha}T^{*}$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(TS)^{*}y\rangle = \langle TSx,y\rangle = \langle Sx,T^{*}y\rangle = \langle x,S^{*}T^{*}y\rangle
      \end{aligned}
      $$
      Therefore, $(TS)^{*} = S^{*}T^{*}$.
    \item We have
      $$
      \begin{aligned}
        \langle x,T^{*}(\alpha u+\beta v)\rangle = \langle T(x), \alpha u+\beta v\rangle &= \overline{\langle \alpha u+\beta v, T(x)\rangle} \\
                                                                                         &= \bar{\alpha}\overline{\langle u,T(x)\rangle}+\bar{\beta}\overline{\langle v,T(x)\rangle} \\
                                                                                         &= \bar{\alpha}\langle T(x),u\rangle+\bar{\beta}\langle T(x),v\rangle \\
                                                                                         &= \bar{\alpha}\langle x,T^{*}(u)\rangle+\bar{\beta}\langle x,T^{*}(v)\rangle \\
                                                                                         &= \langle x,\bar{\alpha}T^{*}(u)+\bar{\beta}T^{*}(v)\rangle
      \end{aligned}
      $$
      Therefore, $T^{*}(\alpha u+\beta v) = \bar{\alpha} T^{*}u+\bar{\beta} T^{*}v$.
  \end{enumerate}
\end{proof}
