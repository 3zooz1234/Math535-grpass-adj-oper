\section{Riesz Representation Theorem}

\begin{thm}[Riesz Representation Theorem]
    Let $H$ be a Hilbert space.
    For any bounded linear functional $\varphi \in H^{*}$,
    there exists a unique element $y \in H$ such that\\

$$\varphi(x)=\langle y, x\rangle \quad$$ for all $x \in H$ and $\|\varphi\|=\|y\|$ \\

\end{thm}

Now recall that $\mathbb{R}^n$, $(\ell_2$ and $L_2[a,b]$ are Hilbert spaces. Using the Riesz representation theorem, we can know the nature of bounded linear functionals on these spaces as in the following examples.

\begin{exm}
    For $F \in (\mathbb{R}^n)^{*}$, there exists a unique $a = (a_1, a_2, a_3, \dots, a_n) \in \mathbb{R}^n$ such that
    \begin{align*}
        F(x) = \langle x, a \rangle = \sum_{i=1}^{n} x_i a_i ,  \text{ for all } x = (x_1, x_2, \dots, x_n) \in \mathbb{R}^n
    \end{align*}
\end{exm}

\begin{exm}
   For $F \in (\ell_2)^{*}$, there exists a unique $a = (a_1, a_2, a_3, \dots, a_n, \dots) \in \ell_2$ such that
    \begin{align*}
        F(x) = \langle x, a \rangle = \sum_{i=1}^{\infty} x_i \overline{a_i}, \text{ for all } x = (x_1, x_2, \dots, x_n, \dots) \in \ell_2 
    \end{align*} 
\end{exm}

\begin{exm}
    For $F \in (L_2[a,b])^{*}$, there exists a unique $g \in L_2[a,b]$ such that
    \begin{align*}
        F(f) = \langle f, g \rangle = \int_{a}^{b} f(t) \overline{g}(t) dt, \text{ for all } f \in L_2 \\
    \end{align*}
\end{exm}

\begin{cor} %File 5
    Let $H$ be a Hilbert space and $T \in B(H)$, then

    $$
    \|T\|  = \sup \{ | \langle T(x), y \rangle | : x, y \in X \text{ and } \|x\| = \|y\| = 1 \}.
    $$
\end{cor}

\begin{proof}
Let $\alpha = \sup \{ | \langle T(x), y \rangle | : x, y \in H \text{ and } \|x\| = \|y\| = 1 \}$.
By the Cauchy-Schwarz inequality, we have $| \langle T(x), y \rangle | \leq \|T(x)\| \|y\|$ and by the definition of the operator norm we get $ \|T(x)\| \|y\| \leq \|T\| \|x\| \|y\|$. Combining these two inequalities and substituting $\|x\| = \|y\| = 1$, we obtain $| \langle T(x), y \rangle | \leq \|T\|$. Taking the supremum, we reach $\alpha \leq \|T\| $.

Now, let $x \in H$ be such that $\|x\| = 1$ and $T(x) \neq 0$. (If $T(x) = 0$ $\forall \|x\| = 1$, then $\|T\|  = 0$ and so $\|T\| \leq \alpha$). As $T(x) \neq 0$, by one of the consequences of the Hahn-Banach Theorem (Corollary 6.1 from the notes), there exists $f \in H^*$ continuous such that $\|f\| = 1$ and $f(T(x)) = \|T(x)\|$. Now since $f \in H^{*}$ and $f$ is bounded, by the Riesz Representation Theorem, there exists $y_f \in H$ such that $\|y_f\| = \|f\| = 1$ and $\|T(x)\| = f(T(x)) = \langle T(x), y_f \rangle$. Therefore, $\|T(x)\| \leq \alpha$ for all $\|x\| = 1$. This implies $\|T\| \leq \alpha$.

Thus, $\|T\|  = \sup \{ | \langle T(x), y \rangle | : x, y \in X \text{ and } \|x\| = \|y\| = 1 \}$
\end{proof}

\section{The Adjoint}

\begin{defn}[Adjoint Operator]
  Let $H$ and $G$ be Hilbert spaces and let $T: H \longrightarrow G$ be a bounded linear operator. An operator $T^{*}: G \rightarrow H$ such that
\begin{equation}
\langle T(x), y\rangle_{G}=\left\langle x, T^{*}(y)\right\rangle_{H} \quad
\end{equation}
 for all  x $\in H, y \in G$, is called the adjoint of $T$.\label{eq:adjoint}
\end{defn}

\begin{thm}
For any bounded linear operator $T$ on a Hilbert space $H$, the adjoint operator $T^{*}$ exists and is unique.
\end{thm}

\begin{proof}
  \textbf{Existence of the Adjoint:} We would like to show that for all $y\in G$, there is $w\in H$ such that 
  $$\langle Tx,y\rangle = \langle x,w\rangle$$
  for all $x\in H$. Let
  $$
  \begin{aligned}
    f_{y}: H&\longrightarrow\mathbb{F} \\
           x&\longmapsto\langle Tx,y\rangle
  \end{aligned}
  $$
  This map is linear, since it is the composition of two linear maps, namely $T$ and $\langle\cdot,y\rangle$. Also, this map is bounded since
  $$
  \begin{aligned}
    |f_{y}(x)| = |\langle Tx,y\rangle| &\leq \lVert y\rVert\lVert Tx\rVert \\
                                       &\leq \lVert y\rVert\lVert T\rVert \lVert x\rVert
  \end{aligned}
  $$
  So, $f_{y}\in H^{*}$. By Reisz Representation Theorem, for some $w\in H$, we can write $f_{y}$ as
  $$
  \langle Tx,y\rangle = f_{y}(x) = \langle x,w\rangle
  $$
  \newline
  \textbf{Uniqueness of the Adjoint:} Let $S:H\to G$ and $S':H\to G$ be adjoints of $T$, i.e. for all $x,y\in G$
  $$
    \begin{aligned}
      \langle Tx,y\rangle &= \langle x,Sy\rangle \\
      \langle Tx,y\rangle &= \langle x,S'y\rangle
    \end{aligned}
  $$
  We then have
  $$
  \begin{aligned}
    \langle x,(S-S')y\rangle = \langle x,Sy\rangle-\langle x,S'y\rangle = \langle Tx,y\rangle-\langle Tx,y\rangle = 0 \\
  \end{aligned}
  $$
  This implies that $(S-S')y=0$, and hence, $S-S'=0\implies S=S'$.
\end{proof}

The defining relation of the adjoint \eqref{eq:adjoint} shows that the action of $A$ on $H_0$ is mimicked by the action of the adjoint $A^{*}$ on $H_1$; this mimicry is visible only through the applicable inner products, so the adjoint itself depends on these inner products. Roughly, if $A$ has some effect, $A^{*}$ preserves the geometry of that effect while acting with a reversed domain and codomain.

\begin{exm}
Let $I$ be the identity operator on a Hilbert space $H$. then $I^{*}=I$
\end{exm}

\begin{exm}
For $k \in C[0,1]$, let $T_{k}: C[0,1] \rightarrow C[0,1]$ be given by $(T_{k}g)(t)=k(t) g(t)$.
Then $\left(T_{k}\right)^{*}=T_{\bar{k}}$.
\end{exm}

\begin{exm}
Let $T_1:H \rightarrow H$ be given by $T_1(x) = \alpha x$ for some scalar $\alpha$. For any $x$ and $y$ in $H$,
$$
\langle T_1(x), y \rangle = \langle \alpha x, y \rangle = \alpha \langle x, y \rangle = \langle x, \bar{\alpha} y \rangle
$$
where the first equality follows from the definition of $T_1$, the second from the linearity in the first component of the inner product, and the third from the conjugate linearity in the second component of the inner product. Comparing with \eqref{eq:adjoint}, the adjoint of $T_1$ is $T_1^{*}(y) = \bar{\alpha}y$. Hence, the adjoint of multiplication by a scalar is multiplication by the conjugate of the scalar.
\end{exm}

From the above example it follows that if $H$ is a real Hilbert space, then $T_1= T_1^{*}$.

\begin{exm} \label{ShiftOp}
Suppose that $S$ and $T$ are the right and left shift operators, respectively, on the sequence space $l^{2}(\mathbb{R})$, defined by
$$
S(x_{1}, x_{2}, x_{3}, \dots) = (0, x_{1}, x_{2}, x_{3}, \dots), \quad T(x_{1}, x_{2}, x_{3}, \dots) = (x_{2}, x_{3}, x_{4}, \dots).
$$
Then $S^{*}=T$, since
$$
\langle T(x), y \rangle = x_{1}(0) + x_{2}{y_{1}} + x_{3}{y_{2}} + x_{4}{y_{3}} + \dots = \langle x, S(y) \rangle.
$$
Also, $T^{*}=S$, since
$$
\langle S(x), y \rangle = (0)y_{1} + x_{1}{y_{2}} + x_{2}{y_{3}} + x_{3}{y_{4}} + \dots = \langle x, T(y) \rangle.
$$
\end{exm}


\begin{thm}
  Let $H,G$ be Hilbert spaces and let $T:H\to G$ and $S:H\to G$ be linear operator, then
  \begin{enumerate}
    \item $\left(T^{*}\right)^{*}=T$
    \item $(T+S)^{*} = T^{*} + S^{*}$
    \item $(\alpha T)^{*} = \bar{\alpha}T^{*}$
    \item $(TS)^{*} = S^{*}T^{*}$
  \end{enumerate}
\end{thm}
\begin{proof}
  \begin{enumerate}
    \item We have
      $$
      \begin{aligned}
        \langle\left(T^{*}\right)^{*}x,y\rangle = \overline{\langle y,\left(T^{*}\right)^{*}x\rangle} = \overline{\langle T^{*}y,x\rangle} = \langle x,T^{*}y\rangle = \langle Tx,y\rangle
      \end{aligned}
      $$
      Therefore, $\left(T^{*}\right)^{*}=T$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(T+S)^{*}y\rangle = \langle (T+S)x,y\rangle &= \langle Tx,y\rangle+\langle Sx,y\rangle\\
                                                              &= \langle x,T^{*}y\rangle+\langle x,S^{*}y\rangle \\
                                                              &= \langle x,(T^{*}+S^{*})y\rangle
      \end{aligned}
      $$
      Therefore, $(T+S)^{*} = T^{*}+S^{*}$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(\alpha T)^{*}y\rangle = \langle\alpha Tx,y\rangle = \alpha\langle Tx,y\rangle = \alpha\langle x,T^{*}y\rangle &= \alpha\overline{\langle T^{*}y,x\rangle} \\
                                                                                                                                 &= \overline{\bar{\alpha}\langle T^{*}y,x\rangle} \\
                                                                                                                                 &= \overline{\langle \bar{\alpha}T^{*}y,x\rangle} \\
                                                                                                                                 &= \langle x,\bar{\alpha}T^{*}y\rangle
      \end{aligned}
      $$
      Therefore, $(\alpha T)^{*} = \bar{\alpha}T^{*}$.
    \item We have
      $$
      \begin{aligned}
        \langle x,(TS)^{*}y\rangle = \langle TSx,y\rangle = \langle Sx,T^{*}y\rangle = \langle x,S^{*}T^{*}y\rangle
      \end{aligned}
      $$
      Therefore, $(TS)^{*} = S^{*}T^{*}$.
  \end{enumerate}
\end{proof}

\begin{prop}
If $H$ is a complex Hilbert space and $T \in B(H)$ is invertible then $T^*$ is invertible and $(T^*)^{-1} = (T^{-1})^*$.
\end{prop}

\begin{proof}
As $TT^{-1} = T^{-1}T = I$, if we take the adjoint of this equation we obtain $(TT^{-1})^* = (T^{-1}T)^* = I^*$ and so $(T^{-1})^*T^* = T^*(T^{-1})^* = I$ (see example below). Hence $T^*$ is invertible and $(T^*)^{-1} = (T^{-1})^*$.
\end{proof}

\begin{thm}
  If $T\in B(H,G)$, then $T^{*}$ is a bounded linear operator with $\lVert T^{*}\rVert = \lVert T\rVert$.
\end{thm}
\begin{proof}
  First, $T^{*}$ is linear, since, for all $x\in H$
  $$
  \begin{aligned}
    \langle x,T^{*}(\alpha u+\beta v)\rangle = \langle T(x), \alpha u+\beta v\rangle &= \overline{\langle \alpha u+\beta v, T(x)\rangle} \\
                                                                                     &= \overline{\alpha\langle u,T(x)\rangle+\beta\langle v,T(x)\rangle} \\
                                                                                     &= \overline{\alpha\langle T^{*}(u),x\rangle+\beta\langle T^{*}(v),x\rangle} \\
                                                                                     &= \overline{\langle\alpha T^{*}(u)+\beta T^{*}(v),x\rangle} \\
                                                                                     &= \langle x,\alpha T^{*}(u)+\beta T^{*}(v)\rangle
  \end{aligned}
  $$
  Now, we show that $T^{*}$ is bounded. Consider the following map
  $$
  \begin{aligned}
    \varphi:H&\longrightarrow\mathbb{F} \\
    x&\longmapsto\langle x,T^{*}y\rangle\\
  \end{aligned}
  $$
  We see that
  $$
  \begin{aligned}
    |\varphi(x)| = |\langle x,T^{*}y\rangle| = |\langle Tx,y\rangle|\leq\lVert y\rVert\lVert Tx\rVert\leq\lVert y\rVert\lVert T\rVert\lVert x\rVert
  \end{aligned}
  $$
  So, $\varphi$ is bounded and $\lVert\varphi\rVert\leq\lVert T\rVert\lVert y\rVert$. By Reisz Representation Theorem, we have $\lVert T^{*}y\rVert=\lVert\varphi\rVert\leq\lVert T\rVert\lVert y\rVert$. Therefore, $T^{*}$ is bounded with $\lVert T^{*}\rVert\leq\lVert T\rVert$. Now, consider the map
  $$
  \begin{aligned}
    \tilde{\varphi}:G&\longrightarrow\mathbb{F} \\
    y&\longmapsto\langle Tx,y\rangle
  \end{aligned}
  $$
  We have
  $$
  |\tilde{\varphi}(y)| = |\langle Tx,y\rangle| = |\langle x,T^{*}y\rangle|\leq\lVert x\rVert\lVert T^{*}y\rVert\leq\lVert x\rVert\lVert T^{*}\rVert\lVert y\rVert
  $$
  So, $\tilde{\varphi}$ is bounded and $\lVert\tilde{\varphi}\rVert\leq\lVert T^{*}\rVert\lVert x\rVert$. By Reisz Representation Theorem, $\lVert Tx\rVert=\lVert\tilde{\varphi}\rVert\leq\lVert T^{*}\rVert\lVert x\rVert$. Therefore, $\lVert T\rVert\leq\lVert T^{*}\rVert$, and hence, $\lVert T^{*}\rVert = \lVert T\rVert$.
\end{proof}

\begin{prop}
  Let $H, G$ be Hilbert spaces, and $T:H \rightarrow G$. then
  \begin{enumerate}
    \item $\left\|T^{*} T\right\|=\|T\|^{2}$
    \item $\operatorname{ker} T=\left(\operatorname{Im} T^{*}\right)^{\perp}$
    \item $\operatorname{ker} T^{*}=(\operatorname{Im}T)^{\perp}$
    \item $\operatorname{ker} T^{*}=\{0\} \Leftrightarrow \operatorname{Im} T$ dense in $G$.
  \end{enumerate}
\end{prop}
\begin{proof}
  \begin{enumerate}
    \item Firstly, we have
      $$
      \lVert T^{*}Tx\rVert\leq\lVert T^{*}\rVert\lVert Tx\rVert\leq\lVert T^{*}\rVert\lVert T\rVert\lVert x\rVert
      $$
      So, $\lVert T^{*}T\rVert\leq\lVert T^{*}\rVert\lVert T\rVert=\lVert T\rVert\lVert T\rVert = \lVert T\rVert^{2}$. We also observe that
      $$
      \begin{aligned}
        \lVert Tx\rVert^{2} = \langle Tx,Tx\rangle = \langle x,T^{*}Tx\rangle \leq \lVert x\rVert\lVert T^{*}Tx\rVert &\leq \lVert x\rVert\lVert T^{*}T\rVert\lVert x\rVert\\ &= \lVert T^{*}T\rVert\lVert x\rVert^{2}
      \end{aligned}
      $$
      So, $\lVert T\rVert^{2}\leq\lVert T^{*}T\rVert$, and therefore, $\left\|T^{*} T\right\|=\|T\|^{2}$
    \item Let $u\in\ker T$, i.e. $Tu=0$. This is true if and only if $\langle Tu,v\rangle=0$ for all $v\in G$. So, $\langle u,T^{*}v\rangle = \langle Tu,v\rangle = 0$. $\langle u,T^{*}v\rangle=0$ if and only if $u$ is orthogonal to every element of $\im T$, i.e. $u\in(\im T^{*})^{\perp}$.
    \item By the second property above, we have
      $$
      \ker T^{*} = (\im (T^{*})^{*})^{\perp} = (\im T)^{\perp}
      $$
    \item If $\ker T^{*} = \{0\}$, then, by the third property above, $(\im T)^{\perp} = \{0\}$. Here, $(\im T)^{\perp}=\{0\}$ if and only if $\im T$ is dense in $G$.
  \end{enumerate}
\end{proof}

\begin{thm} \label{Inv}
Let $H$ be a Hilbert space and $T: H \rightarrow H$ then $T$ is invertible $\Longleftrightarrow$  $\operatorname{ker}T^{*}=\{0\}$ and $\exists \alpha>0$ such that $\|T(x)\| \geqslant \alpha\|x\| \quad \forall x \in H$.
\end{thm}

\begin{exm} \label{ShiftOpInv}
Let $T: l^{2} \rightarrow l^{2}$ where 

$$\begin{aligned} T\left(x_{1}, x_{2} \ldots\right) & =\left(0, x_{1}, x_{2} \ldots\right)   \end{aligned}$$

As shown in Example \ref{ShiftOp}, we have $T^{*}(x_{1},x_{2},\dots) = (x_{2},x_{3},\dots)$. So, $$\operatorname{ker}T^{*} = \{u\in\mathbb{F}: (u,0,0,\dots)\}\not=\{(0,0,0,\dots)\}$$ By Theorem \ref{Inv}, $T$ is not invertible.
\end{exm}

\begin{prop}
Let $H$ be a complex Hilbert space and let $T \in B(H)$, then $\text{Ker } T = \text{Ker } (T^*T)$.
Moreover, the closure of $\text{Im } T^*$ is equal to the closure of $\text{Im } (T^*T)$, where $\text{Im } T^*$ denotes the image of $T^*$.
\end{prop}


\begin{thm}
Suppose that $A:\mathcal{H}\rightarrow\mathcal{H}$ is a bounded linear operator on a Hilbert space $\mathcal{H}$ with closed range. Then the equation $Ax=y$ has a solution for $x$ if and only if $y$ is orthogonal to $\ker A^{*}$.
\end{thm}

\begin{exm}
Consider the multiplication operator $M:L^{2}([0,1])\rightarrow L^{2}([0,1])$ defined by
$$
Mf(x)=xf(x).
$$
Then $M^{*}=M$, and $M$ is one-to-one, so every $g\in L^{2}([0,1])$ is orthogonal to $\ker M^{*}$; but the range of $M$ is a proper dense subspace of $L^{2}([0,1])$, so $Mf=g$ is not solvable for every $g\in L^{2}([0,1])$.
\end{exm}

\section{Normal Operators}

\textbf{\textit{Definition 2}}. Let $H$ be a Hilbert space and $T$ a lin. operaton $T: H \rightarrow H$. then if $T T^{*}=T^{*} T, T$ is said to be normal of eratu.

\begin{exm}
  Consider
  $$
  \begin{aligned}
    T:\mathbb{R}^{2}&\longrightarrow\mathbb{R}^{2}\\
    x&\longmapsto\begin{pmatrix}1&-1\\1&1\\\end{pmatrix}x
  \end{aligned}
  $$
  We see that
  $$
    T^{*}(x) = \begin{pmatrix}1&-1\\1&1\\\end{pmatrix}^{T}x = \begin{pmatrix}1&1\\-1&1\\\end{pmatrix}x
  $$
  Since
  $$
    \begin{pmatrix}1&1\\-1&1\end{pmatrix}\begin{pmatrix}1&-1\\1&1\\\end{pmatrix} = \begin{pmatrix}2&0\\0&2\\\end{pmatrix} = \begin{pmatrix}1&-1\\1&1\\\end{pmatrix}\begin{pmatrix}1&1\\-1&1\\\end{pmatrix}
  $$
  we have $T^{*}T=TT^{*}$.
\end{exm}

\begin{exm}
  If $T$ is the right shift operator defined in Example \ref{ShiftOp}, then $T$ is not normal. This can be shown by finding $T^{*}T$ and $TT^{*}$. We have
  $$
  \begin{aligned}
    T^{*}T(x_{1},x_{2},\dots) = T^{*}(0,x_{1},x_{2},\dots) = (x_{1},x_{2},\dots)
  \end{aligned}
  $$
  and
  $$
  \begin{aligned}
    TT^{*}(x_{1},x_{2},\dots) = T(x_{2},x_{3},\dots) = (0,x_{2},x_{3},\dots)
  \end{aligned}
  $$
  which are not equal.
\end{exm}

\begin{exm}
If $T$ is normal, then $T$ - $\lambda I$ is also normal. That is, since
$$
\begin{aligned}
  (T-\lambda I)^{*}(T-\lambda I) = (T^{*}-\bar{\lambda}I)(T-\lambda I) &= T^{*}T-\bar{\lambda}T-\lambda T^{*}+\bar{\lambda}\lambda I \\
                                                                       &= TT^{*}-\bar{\lambda}T-\lambda T^{*}+\lambda\bar{\lambda}I \\
                                                                       &= T(T^{*}-\bar{\lambda}I) - \lambda (T^{*}-\bar{\lambda}I) \\
                                                                       &= (T-\lambda I)(T^{*}-\bar{\lambda}I) \\
                                                                       &= (T-\lambda I)(T-\lambda I)^{*}
\end{aligned}
$$
\end{exm}

\begin{lem}
Let $T: H \rightarrow H$ be a normal operator, then
  \begin{enumerate}
    \item $\|T(x)\|=\left\|T^{*}(x)\right\| ; \quad \forall x \in H$
    \item If $\|T(x)\| \geqslant \alpha\|x\| \quad \forall x \in H$ then $\operatorname{ker} T^{*}=\{0\}$. ($\alpha>0$ ????)
  \end{enumerate}
\end{lem}
\begin{proof}
  \begin{enumerate}
    \item We have
      $$
      \begin{aligned}
        \lVert Tx\rVert = \langle Tx,Tx\rangle = \langle x,T^{*}Tx\rangle = \langle x,TT^{*}x\rangle = \langle T^{*}x,T^{*}x\rangle = \lVert T^{*}x\rVert
      \end{aligned}
      $$
    \item Let $u\in\operatorname{ker}T^{*}$, i.e. $T^{*}(u)=0$. By 1, we have
      $$
      0=\lVert T^{*}u\rVert=\lVert Tu\rVert\geq\alpha\lVert u\rVert\geq 0
      $$
      Hence, $\lVert u\rVert=0\implies u=0$
  \end{enumerate}
\end{proof}

By the lemma above and theorem \ref{Inv}, the next theorem immediately follows.

\begin{thm}
Let $H$ be a Hilbert space and $T: H \rightarrow H$ is a normal operator, then
$T$ is invertible $\Longleftrightarrow \exists \alpha>$ such thar $\quad\|T(x)\| \geqslant \alpha\|x\| \forall x \in H$.
\end{thm}

Hence it is easier to determine whether a normal operator invertible compared with any operator.

\section{Self-adjoint operators}
An interesting subset of the normal operators is the self-adjoint operators.

\begin{defn}
$T: H \rightarrow H$ is called a self-adjoint if $T=T^{*}$
\end{defn}

\begin{exm}
For any $T \in B(H)$, we observe that $T T^{*}$ and $T^{*} T$ are self-adjoint, since
$$
(T^{*}T)^{*} = T^{*}(T^{*})^{*} = T^{*}T
$$
and
$$
(TT^{*})^{*} = (T^{*})^{*}T^{*} = TT^{*}
$$
\end{exm}

\begin{thm}
Let $H$ be Hlibert space \& $T_{1}$ and $T_{2}$ be self-adjont oterator, then
  \begin{enumerate}
    \item $\alpha T_{1}+\beta T_{2}$ is a self-adjon, $\alpha \beta \in \mathbb{R}$
    \item The set
      $$
      \mathfrak{S} = \{T\in B(H): T\text{ is self-adjoint}\}\subset B(H)
      $$
      is a closed subset of $B(H)$.
  \end{enumerate}
\end{thm}
\begin{proof}
  \begin{enumerate}
    \item We have
      $$
      (\alpha T_{1}+\beta T_{2})^{*} = (\alpha T_{1})^{*} + (\beta T_{2})^{*} = \bar{\alpha} T_{1}^{*}+\bar{\beta} T_{2}^{*} = \alpha T_{1}^{*}+\beta T_{2}^{*} = \alpha T_{1}+ \beta T_{2}
      $$
      Hence, $\alpha T_{1}+\beta T_{2}$ is self-adjoint.
    \item Let $\{T_{n}\}$ be a convergent sequence of elements in $\mathfrak{S}$ which converges to $T\in B(H)$, then
      $$
      \lim_{n\to\infty}\lVert T_{n}-T\rVert = 0
      $$
      We see that
      $$
      \lim_{n\to\infty}\lVert T_{n}-T^{*}\rVert = \lim_{n\to\infty}\lVert T_{n}^{*}-T^{*}\rVert = \lim_{n\to\infty}\lVert (T_{n}-T)^{*}\rVert = \lim_{n\to\infty}\lVert T_{n}-T\rVert = 0
      $$
      So,
      $$
      \lim_{n\to\infty}\lVert T-T^{*}\rVert\leq\lim_{n\to\infty}\lVert T_{n}-T\rVert+\lim_{n\to\infty}\lVert T_{n}-T^{*}\rVert = 0+0 = 0
      $$
      Hence, $T^{*}=T$ and $T\in\mathfrak{S}$. Therefore, $\mathfrak{S}\subset B(H)$ is a closed set.
  \end{enumerate}
\end{proof}

\begin{prop}
Any operator $T \in B(H)$, $T=R+i S$ for $R$ and $S$ self-adjoint
\end{prop}
\begin{proof}
  Let $R$ and $S$ be the following
  $$
  R=\frac{T+T^{*}}{2}\quad S=\frac{T-T^{*}}{2i}
  $$
  then we have
  $$
  R+iS = \Big(\frac{T+T^{*}}{2}\Big)+i\Big(\frac{T-T^{*}}{2i}\Big) = \frac{T+T^{*}}{2}+\frac{T-T^{*}}{2} = \frac{2T}{2} = T
  $$
  and
  $$
  \begin{aligned}
    R^{*} = \Big(\frac{T+T^{*}}{2}\Big)^{*} = \frac{(T+T^{*})^{*}}{2} = \frac{T^{*}+(T^{*})^{*}}{2} = \frac{T^{*}+T}{2} = \frac{T+T^{*}}{2} = R \\ \\
    S^{*} = \Big(\frac{T-T^{*}}{2i}\Big)^{*} = -\frac{(T-T^{*})^{*}}{2i} = -\frac{T^{*}-(T^{*})^{*}}{2i} = -\frac{T^{*}-T}{2i} = \frac{T-T^{*}}{2i} = S
  \end{aligned}
  $$
\end{proof}

\section{Unitary operators}

\begin{defn}
An operator $T \in B(H)$ called unitary, if $TT^{*}=T^{*}T=I$.
\end{defn}

\begin{exm} \label{IdIsUnitary}
  It can be see that the identity operator, $I$, is unitary, since $I^{*}I=II=I=II=II^{*}$.
\end{exm}

\begin{prop}
Let $T \in B(H)$, then $T$ is unitary if $T$ is isometry.
\end{prop}
\begin{proof}
  Let $T\in B(H)$ be an isometry, we have
  $$
  \begin{aligned}
    \langle x,T^{*}Tx\rangle = \langle Tx,Tx\rangle = \lVert Tx\rVert &= \lVert x\rVert = \langle x,x\rangle = \langle x,Ix\rangle\\
                                                                      &\implies \langle x,(T^{*}T-I)x\rangle = 0
  \end{aligned}
  $$
  Hence, $(T^{*}T-I)x = 0$ for all $x\in H$. This implies that $T^{*}T-I = 0\implies T^{*}T = I$. Therefore, $T$ is unitary.
\end{proof}

\begin{thm} \label{UnitaryProp}
Let $T, S$ be unitary operators, then
\begin{enumerate}
  \item $T^{*}$ is unitary and $\|T\|=\left\|T^{*}\right\|=1$
  \item $T S$ is unitary
  \item $T^{-1}$ is unitary.
\end{enumerate}
\end{thm}
\begin{proof}
  \begin{enumerate}
    \item We know that $T^{*}T = I\implies T^{-1}=T^{*}$. So,
      $$
      (T^{*})^{-1}=(T^{-1})^{*}=(T^{*})^{*}\implies (T^{*})^{*}T^{*}=I
      $$
    \item We have
      $$
      \begin{aligned}
        (TS)^{*}TS = S^{*}T^{*}TS = S^{*}IS = S^{*}S = I
      \end{aligned}
      $$
    \item We have $(T^{-1})^{*}T^{-1} = (T^{*})^{*}T^{-1} = TT^{-1} = I$
  \end{enumerate}
\end{proof}

\begin{rem}
  By Theorem \ref{UnitaryProp}, Example \ref{IdIsUnitary} and the associativity of function composition, we observe that the set of all unitary operators forms a group under composition. Moreover, if $T:\mathbb{C}^{n}\to\mathbb{C}^{n}$, this group is called the \textbf{unitary group}, denoted $U(n)$. There is a special subgroup of $\operatorname{U}(n)$, which is the \textbf{special unitary group}, denoted $\operatorname{SU}(n)$, which is the set of all unitary operators from $\mathbb{C}^{n}$ to $\mathbb{C}^{n}$ such that its determinant is $1$. This forms a subgroup since if we let $A,B\in\operatorname{SU}(n)$, we have
  $$
  \operatorname{det}AB^{-1} = \frac{\operatorname{det}A}{\operatorname{det}B} = \frac{1}{1} = 1
  $$
  showing that $AB^{-1}\in\operatorname{SU}(n)$.
\end{rem}

\section{The Spectrum of an Operator}
The notation of an eigenvalue of a matrix has many essential applications in finite dimensional vector space; however, we wish we could extend this notation to the infinite-dimensional vector space.

\begin{defn}
Let $T \in B(H), \lambda \in \mathbb{C}$ is called an eigenvalue of $T$, if there exists $x \in H$ with $x\not=0$ such that

$$
Tx=\lambda x
$$
\end{defn}

\begin{defn}
  Let $T \in B(H)$. The resolvent set of $T$, denoted by $\rho(T)$, is $\rho(T) = \{\lambda\in\mathbb{C}: T-\lambda I\text{ is invertible}\}$. The spectrum of $T$, denoted by $\sigma(T)$ is $\sigma(T)=\mathbb{C}\backslash\rho(T)$, or equivalently, $\sigma(T) = \{\lambda\in\mathbb{C}: T-\lambda I\text{ is not invertible}\}$
\end{defn}

\begin{rem} \label{defRho}
  Notice that, by Theorem \ref{Inv}, we can rewrite the resolvent set of $T$ as
  $$
  \rho(T) = \{\lambda\in\mathbb{C}: \operatorname{ker}(T-\lambda I)^{*}=\{0\}\text{ and }\lVert (T-\lambda I)x\rVert\geq\alpha\lVert x\rVert\text{ for some }\alpha>0\}
  $$
\end{rem}

\begin{thm}
If $\lambda$ is an eigenvalue of $T$ , then $\lambda \in \sigma(T)$
\end{thm}
\begin{proof}
  Suppose that $\lambda\notin\sigma(T)$, then $T-\lambda I$ is invertible. So, we have $(T-\lambda I)u=0\implies u = (T-\lambda I)^{-1}(T-\lambda I)u = (T-\lambda I)^{-1}(0)=0$. So, if there is $u\in\mathcal{H}$ such that $Tu=\lambda u$, then $u=0$. So, $\lambda$ is not an eignvalue of $T$.
\end{proof}

\begin{thm}
  Let $H$ be a finite dimensional Hilbert space, and let $T:H\to H$. If $\lambda\in\sigma(T)$, then $\lambda$ is an eigenvalue of $T$.
\end{thm}
\begin{proof}
  Suppose that $\lambda$ is not an eigenvalue of $T$, then there is nonzero $u\in H$ such that $Tu=\lambda u$. So, there is no nonzero $u\in H$ such that $(T-\lambda I)u=0$. So, $\ker(T-\lambda I)=\{0\}$. By rank-nullity theorem, we have
  $$
  \begin{aligned}
    \dim H &= \dim\ker(T-\lambda I) + \dim\im(T-\lambda I) \\
           &= 0 + \dim\im(T-\lambda I) \\
           &= \dim\im(T-\lambda I)
  \end{aligned}
  $$
  So, $\im(T-\lambda I)=H$. This means that $(T-\lambda I)$ is surjective, and therefore, invertible. So, $\lambda\in\rho(T)$, and therefore, $\lambda\notin\sigma(T)$.
\end{proof}

\begin{rem}
For finite dimesonial Hilbert space, the spectrum of $T$ consist only the eigenvalues of $T$. However, in infite dimesional Hilbert space, we may have no eigenvalues. For instance, consider the shift operator defined in Example \ref{ShiftOp}. 
\end{rem}

\begin{thm} \label{SpectIsBd}
Let $\mathcal{H}$ be a complex Hilbert space and let $T \in B(\mathcal{H})$. If $|\lambda|>\|T\|$ then $\lambda \notin \sigma(T)$.
\end{thm}
\begin{proof}
  Let $u\in\operatorname{ker}(T-\lambda I)^{*}$, then $(T-\lambda I)^{*}u = (T^{*}-\bar{\lambda}I)u = 0$. So, $T^{*}u-\bar{\lambda}u = 0\implies T^{*}u = \bar{\lambda}u$. So,
  $$
  \begin{aligned}
    |\lambda|\lVert u\rVert = |\bar{\lambda}|\lVert u\rVert = \lVert\bar{\lambda}u\rVert = \lVert T^{*}u\rVert&\leq\lVert T^{*}\rVert\lVert u\rVert = \lVert T\rVert\lVert u\rVert\\\\
                                                                                                              &\implies (|\lambda|-\lVert T\rVert)\lVert u\rVert\leq 0
  \end{aligned}
  $$
  Since $|\lambda|>\lVert T\rVert$, then $\lVert u\rVert\leq 0\implies \lVert u\rVert = 0\implies u=0$. Hence, $\operatorname{ker}(T-\lambda I)^{*} = \{0\}$. We also have that
  $$
  \begin{aligned}
    \lVert (T-\lambda I)x\rVert = \rVert Tx-\lambda x\lVert\geq|\lVert Tx\rVert - |\lambda|\lVert x\rVert| &= \Big|\frac{\lVert Tx\rVert}{\lVert x\rVert} - |\lambda|\Big|\lVert x\rVert\\
                                                                                                           &= \Big(|\lambda|-\frac{\lVert Tx\rVert}{\lVert x\rVert}\Big)\lVert x\rVert\\
                                                                                                           &\geq (|\lambda|-\lVert T\lVert)\lVert x\rVert\\
                                                                                                           &= \alpha\lVert x\rVert
  \end{aligned}
  $$
  where $\alpha:=|\lambda|-\lVert T\rVert$ is positive, since $|\lambda|>\lVert T\rVert$. So, $\lambda\in\rho(T)$, by Remark \ref{defRho}. Hence, $\lambda\notin\sigma(T)$.
\end{proof}

\begin{thm} \label{SpectIsCl}
  $\sigma(T)\subseteq\mathbb{C}$ is a closed set.
\end{thm}
\begin{proof}
  We will show that $\rho(T)=\mathbb{C}\backslash\sigma(T)$ is open. Let $\lambda\in\rho(T)$. This means that, by Remark \ref{defRho}, there is $\alpha>0$ such that
  $$
  \lVert(T-\lambda I)u\rVert\geq\alpha\lVert u\rVert\implies\frac{\lVert(T-\lambda I)u\rVert}{\lVert u\rVert}\geq\alpha
  $$
  Since the quantity $\frac{\lVert(T-\lambda I)u\rVert}{\lVert u\rVert}$ has a lower bound, its infimum exists. Now, let
  $$
  \gamma:=\inf_{x\in H\backslash\{0\}}\frac{\lVert(T-\lambda I)u\rVert}{\lVert u\rVert}
  $$
  and let $\delta:=\frac{\gamma}{2}$. We want to show that
  $$
  B(\lambda,\delta)\subseteq\rho(T)
  $$
  Let $\tilde{\lambda}\in B(\lambda,\delta)$, then $|\tilde{\lambda}-\lambda|<\delta$. First, we observe that
  $$
  \begin{aligned}
    \lVert(T-\tilde{\lambda}I)u\rVert=\lVert(T-\lambda I)u-(\tilde{\lambda}-\lambda)I(u)\rVert&\geq|\lVert(T-\lambda I)u\rVert-|\bar{\lambda}-\lambda|\lVert u\rVert| \\
                                                                                              &=\Big|\frac{\lVert(T-\lambda I)u\rVert}{\lVert u\rVert}-|\tilde{\lambda}-\lambda|\Big|\lVert u\rVert \\
                                                                                              &=\Big(\frac{\lVert(T-\lambda I)u\rVert}{\lVert u\rVert}-|\tilde{\lambda}-\lambda|\Big)\lVert u\rVert \\
                                                                                              &>(\gamma-\frac{\gamma}{2})\lVert u\rVert \\
                                                                                              &=\frac{\gamma}{2}\lVert u\rVert
  \end{aligned}
  $$
  Also, we claim that $\ker(T-\tilde{\lambda}I)^{*}=\{0\}$. Suppose for contradiction that there exists $v\in H\backslash\{0\}$ such that $(T-\tilde{\lambda}I)^{*}v=0$. We, now, have
  $$
  \begin{aligned}
    \lVert(T-\lambda I)v\rVert=\lVert(T-\lambda I)^{*}v+(\overline{\lambda-\tilde{\lambda}})v\rVert&\leq\lVert(T-\lambda I)^{*}v\rVert+|\tilde{\lambda}-\lambda|\lVert v\rVert \\
                                                                                                   &=|\tilde{\lambda}-\lambda|\lVert v\rVert \\
                                                                                                   &<\delta\lVert v\rVert \\
                                                                                                   &=\frac{\gamma}{2}\lVert v\rVert \\
  \end{aligned}
  $$
  Since $v\not=0$, dividing both sides of the inequality yields
  $$
  \gamma\leq\frac{\lVert(T-\lambda I)v\rVert}{\lVert v\rVert}<\frac{\gamma}{2}
  $$
  which is a contradiction. So, $v$ must be $0$ and $\ker(T-\tilde{\lambda}I)=\{0\}$. Hence, $\tilde{\lambda}\in\rho(T)$, and therefore, $B(\lambda,\delta)\subseteq\rho(T)$.
\end{proof}

\begin{lem} \label{SpectBar}
If $\mathcal{H}$ is a complex Hilbert space and $T \in B(\mathcal{H})$ then $\sigma\left(T^{*}\right)=\{\bar{\lambda}: \lambda \in \sigma(T)\}$.
\end{lem}

\begin{exm}
  Let $T:\ell^{2}\to\ell^{2}$ be the unilateral shift operator defined in Example \ref{ShiftOp}. We would like to find the spectrum of $T$, $\sigma(T)$. First, we claim that if $\lambda\in\mathbb{C}$ such that $|\lambda|<1$, then $\lambda$ is an eigen value of $T^{*}$. Consider the sequence
  $$
  x = (1,\lambda,\lambda^{2},\dots)
  $$
  Here, $x\in\ell^{2}$, since
  $$
  \begin{aligned}
    \sum^{\infty}_{n=0}|x_{n}|^{2} = \sum^{\infty}_{n=0}(|\lambda|^{n})^{2} = \sum^{\infty}_{n=0}(|\lambda|^{2})^{n} = \frac{1}{1-|\lambda|^{2}} < \infty
  \end{aligned}
  $$
  where this series converge, since $|\lambda|<1$. Now, observe that
  $$
  \begin{aligned}
    Tx &= T(1,\lambda,\lambda^{2},\dots) \\
       &= (\lambda,\lambda^{2},\lambda^{3},\dots) = \lambda(1,\lambda,\lambda^{2},\dots) = \lambda x
  \end{aligned}
  $$
  So, $x\in\ell^{2}$ is an eigenvalue of $T$. This means that $\{\lambda\in\mathbb{C}:|\lambda|<1\}\subseteq\sigma(T^{*})$. By lemma \ref{SpectBar}, we have $\{\lambda\in\mathbb{C}:|\lambda|<1\}=\{\bar{\lambda}:|\lambda|<1\}\subseteq\{\bar{\lambda}:\lambda\in\sigma(T^{*})\}=\sigma(T)$. So, by theorem \ref{SpectIsBd} and that $T$ is an isometry, we have
  $$
  \{\lambda\in\mathbb{C}:|\lambda|<1\}\subseteq\sigma(T)\subseteq\{\lambda\in\mathbb{C}:|\lambda|\leq 1\}
  $$
  By theorem \ref{SpectIsCl}, $\sigma(T)$ is closed. Therefore,
  $$
  \sigma(T)=\{\lambda\in\mathbb{C}:|\lambda|\leq 1\}
  $$
\end{exm}
